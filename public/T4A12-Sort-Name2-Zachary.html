import java.awt.*; import java.awt.event.*; import javax.swing.*; public class RandomNumberSorter extends JFrame { private JTextArea numberArea; private JButton randomButton, sortButton; public RandomNumberSorter() { super("Random Number Sorter"); setLayout(new BorderLayout()); numberArea = new JTextArea(10, 20); add(new JScrollPane(numberArea), BorderLayout.CENTER); JPanel buttonPanel = new JPanel(); randomButton = new JButton("Generate Random Numbers"); randomButton.addActionListener(new ActionListener() { public void actionPerformed(ActionEvent e) { generateNumbers(); } }); buttonPanel.add(randomButton); sortButton = new JButton("Sort Numbers"); sortButton.addActionListener(new ActionListener() { public void actionPerformed(ActionEvent e) { sortNumbers(); } }); buttonPanel.add(sortButton); add(buttonPanel, BorderLayout.SOUTH); setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); pack(); setVisible(true); } private void generateNumbers() { int[] numbers = new int[3]; for (int i = 0; i < numbers.length; i++) { numbers[i] = (int) (Math.random() * 100); } displayNumbers(numbers); } private void sortNumbers() { String[] lines = numberArea.getText().split("\\n"); int[] numbers = new int[lines.length]; for (int i = 0; i < lines.length; i++) { numbers[i] = Integer.parseInt(lines[i]); } java.util.Arrays.sort(numbers); displayNumbers(numbers); } private void displayNumbers(int[] numbers) { numberArea.setText(""); for (int i = 0; i < numbers.length; i++) { numberArea.append(numbers[i] + "\n"); } } public static void main(String[] args) { new RandomNumberSorter(); } }
